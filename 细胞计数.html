    <script>

        document.addEventListener('DOMContentLoaded', function() {

            const calculateBtn = document.getElementById('calculateBtn');

            const resetBtn = document.getElementById('resetBtn');

            const resultDiv = document.getElementById('result');

            

            // 计算按钮点击事件

            calculateBtn.addEventListener('click', function() {

                // 获取输入值

                const a = parseInt(document.getElementById('a').value);

                const b = parseFloat(document.getElementById('b').value);

                const c = parseInt(document.getElementById('c').value);

                const d = parseInt(document.getElementById('d').value);

                const e = parseFloat(document.getElementById('e').value);

                const f = parseInt(document.getElementById('f').value);

                

                // 验证输入

                if (isNaN(a) || isNaN(b) || isNaN(c) || isNaN(d) || isNaN(e) || isNaN(f)) {

                    alert('请填写所有字段！');

                    return;

                }

                

                if (b <= 0) {

                    alert('细胞悬液体积必须大于0！');

                    return;

                }

                

                // 计算活细胞总数

                const z = a/4 * Math.pow(10, 4) * c * b;

                

                // 计算所需体积

                const allneedvolumn_mL = e * f / 1000;

                const allneedcells = f * d;

                const formerC_mL = z / b;

                const needformerC_mL = allneedcells / formerC_mL;

                const needsolventvolumn_mL = allneedvolumn_mL - needformerC_mL;

                

                // 显示结果

                document.getElementById('totalCells').textContent = 

                    '该溶液中的活细胞总数: ' + formatNumber(z) + ' 个';

                document.getElementById('originalVolume').textContent = 

                    '需要添加的原有细胞悬液体积: ' + formatNumber(needformerC_mL, 4) + ' mL';

                document.getElementById('solventVolume').textContent = 

                    '需要添加的溶剂体积: ' + formatNumber(needsolventvolumn_mL, 4) + ' mL';

                

                resultDiv.style.display = 'block';

                

                // 滚动到结果区域

                resultDiv.scrollIntoView({ behavior: 'smooth' });

            });

            

            // 重置按钮点击事件

            resetBtn.addEventListener('click', function() {

                // 清空所有输入

                document.getElementById('a').value = '';

                document.getElementById('b').value = '';

                document.getElementById('c').value = '';

                document.getElementById('d').value = '';

                document.getElementById('e').value = '';

                document.getElementById('f').value = '';

                

                // 隐藏结果

                resultDiv.style.display = 'none';

                

                // 滚动到顶部

                window.scrollTo({ top: 0, behavior: 'smooth' });

            });

            

            // 数字格式化函数

            function formatNumber(num, decimals = 2) {

                if (num >= 1000000) {

                    return (num / 1000000).toFixed(decimals) + ' × 10^6';

                } else if (num >= 1000) {

                    return (num / 1000).toFixed(decimals) + ' × 10^3';

                } else {

                    return num.toFixed(decimals);

                }

            }

        });
